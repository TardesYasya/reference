// Ассиметричная генерация одинакогово ключа для ключей шифрования

p = случайное простое число
g = простое, первообразный корень по модулю p
        (g^Ф(p)) mod p = 1
        (g^(p - 1)) mod p = 1

a и b = случайные натуральные числа пользователей - приватные ключи

k1 = (g^a) mod p
k2 = (g^b) mod p

(k1^b) mod p == (k2^a) mod p
        
Пример
    g = 3
    p = 17
    (3^x) mod 17


    Случайное число 1 = 84
    (3^84) mod 17 = 13
    
    Случайное число 2 = 67
    (3^67) mod 17 = 10
    
    Создание ключей
    
    Ключ 1
        (10^84) mod 17 = 4
        
    Ключ 2
        (13^67) mod 17 = 4
