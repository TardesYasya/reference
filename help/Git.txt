# CMD

# Статус проекта
git status

# История изменений
git log

# Внесённые изменения
git diff


# Скачать репозиторий
git clone https://github.com/homaander/test.git

# Обновить репозиторий
git pull


# Подготовка отправки одного или всех файлов к отправке
git add "index.html"
git add -A

# Добовление коментария
git commit -m "New index file"

# Вход в аккаунт
git config --global user.email "you@example.com"
git config --global user.name "Your Name"

# Отправка проекта
git push

# 
git remote show origin

# Все ветки проекта
git branch

# Создать ветку dev
git branch dev

# Удалить ветку dev
git branch -d dev

# Изменить ветку dev
git checkout dev

# Присоединение ветки
git merge dev

# Просмотр истории
gitk
