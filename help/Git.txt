
git status
	:: Статус проекта
git log
	:: История изменений
git diff
	:: Внесённые изменения


git add "index.html"
git add -A
	:: Подготовка отправки одного или всех файлов

git clone https://github.com/homaander/test.git

git commit -m "New index file"
	:: Добовление коментария

git config --global user.email "you@example.com"
git config --global user.name "Your Name"

git push
	:: Отправка проекта

git pull
	:: Обновить файлы с облока


git remote show origin

git branch
	:: Все ветки проекта

git branch dev
	:: Создать ветку
git branch -d dev
	:: Удалить ветку
git checkout dev
	:: Изменить ветку

git merge dev
	:: Присоединение ветки 


gitk
	:: Просмотр истории
